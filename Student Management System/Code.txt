<script>
$(document).ready(function () {

    // Accordion toggle â€” click the parent <li> trigger link
    $(".sidebar_menu .menu-toggle").on("click", function (e) {
        e.preventDefault();
        var $li = $(this).closest("li");
        var isOpen = $li.hasClass("active");

        // Close all open submenus
        $(".sidebar_menu li.has-submenu.active").not($li).removeClass("active")
            .find(".accordion").slideUp(250);

        // Toggle clicked item
        if (isOpen) {
            $li.removeClass("active");
            $li.find(".accordion").slideUp(250);
        } else {
            $li.addClass("active");
            $li.find(".accordion").slideDown(250);
        }
    });

    // Highlight active page
    var currentPath = window.location.pathname.toLowerCase();
    $(".sidebar_menu a").each(function () {
        if ($(this).attr("href") && currentPath.indexOf($(this).attr("href").toLowerCase().replace("../", "").replace(".aspx", "")) > -1) {
            $(this).closest("li").addClass("current");
        }
    });

    // Hamburger open
    $(".hamburger").on("click", function () {
        $(".wrapper").addClass("sidebar-open");
    });

    // Close button & overlay close
    $(".close, .bg_shadow").on("click", function () {
        $(".wrapper").removeClass("sidebar-open");
    });
});
</script>
